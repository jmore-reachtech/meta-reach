bootcmd=setenv panel ${board_rev}; run check_update;
baudrate=115200
loadaddr=0x12000000
fdt_addr=0x18000000
image=zImage
fdtfile=devicetree.dtb
rescue_image=rescue-image.ext3.gz
ip_dyn=yes
splashpos=m,m
console=ttymxc0
mmcdev=0
mmcpart=1
mmcroot=/dev/mmcblk0p2 rootwait rw rootfstype=ext4
mmcargs=setenv bootargs console=${console},${baudrate} quiet root=${mmcroot} consoleblank=0 vt.global_cursor_default=0
loadimage=fatload mmc ${mmcdev}:${mmcpart} ${loadaddr} ${image}
loadfdt=fatload mmc ${mmcdev}:${mmcpart} ${fdt_addr} ${fdtfile}
mmcboot=echo Booting from mmc ...; run mmcargs; run loadimage; run loadfdt; bootz ${loadaddr} - ${fdt_addr};
netargs=setenv bootargs console=${console},${baudrate} quiet root=/dev/nfs ip=dhcp nfsroot=${serverip}:${nfsroot},v3,tcp
netboot=echo Booting from net ...; run netargs; if test ${ip_dyn} = yes; then setenv get_cmd dhcp; else setenv get_cmd tftp; fi; ${get_cmd} ${image}; if test ${boot_fdt} = yes || test ${boot_fdt} = try; then if ${get_cmd} ${fdt_addr} ${fdtfile}; then bootz ${loadaddr} - ${fdt_addr}; else if test ${boot_fdt} = try; then bootz; else echo WARN: Cannot load the DT; fi; fi; else bootz; fi;
nandargs=setenv bootargs console=${console},${baudrate} quiet ubi.mtd=2,2048 root=ubi0:rootfs0 rootfstype=ubifs consoleblank=0 vt.global_cursor_default=0
nandboot=echo Booting from nand ...; run nandargs; nand read ${fdt_addr} 0x0000000 0x0080000; nand read ${loadaddr} 0x0080000 0x0A00000; bootz ${loadaddr} - ${fdt_addr};  
bootdelay=0
swupdate=false
rescue_addr=0x19000000
load_sd_rescue_image=fatload mmc ${mmcdev}:${mmcpart} ${rescue_addr} ${rescue_image}
load_nor_rescue_image=sf probe; sf read 0x19000000 0x000E0000 0x800000; 
rescue_root=/dev/ram0 rw initrd=0x19000000,8M
rescue_args=setenv bootargs console=${console},${baudrate} init=/sbin/init root=${rescue_root} consoleblank=0 vt.global_cursor_default=0
rescue_sd_boot=echo Rescue boot ...; run rescue_args; run load_sd_rescue_image; run loadimage; run loadfdt; bootz ${loadaddr} - ${fdt_addr};
rescue_nor_boot=echo Rescue boot ...; run rescue_args; run load_nor_rescue_image; nand read ${fdt_addr} 0x0000000 0x0080000; nand read ${loadaddr} 0x0080000 0x0A00000; bootz ${loadaddr} - ${fdt_addr};
check_update=if test ${swupdate} = true; then run rescue_nor_boot; else run nandboot; fi;
