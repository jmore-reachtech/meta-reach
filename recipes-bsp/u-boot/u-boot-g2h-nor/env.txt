baudrate=115200
bootcmd=sf probe; run loadnorimage; run norboot;
bootdelay=1
console=ttymxc0
ethact=FEC
ethprime=FEC
fdt_addr=0x18000000
fdt_file=devicetree.dtb
image=zImage
ip=dhcp nfsroot=${serverip}:${nfsroot},v3,tcp
ip_dyn=yes
loadaddr=0x12000000
loadfdt=fatload mmc ${mmcdev}:${mmcpart} ${fdt_addr} ${fdt_file}
loadimage=fatload mmc ${mmcdev}:${mmcpart} ${loadaddr} ${image}
mmcargs=setenv bootargs console=${console},${baudrate} root=${mmcroot} consoleblank=0 vt.global_cursor_default=0 quiet
mmcboot=echo Booting from mmc ...; run mmcargs; if run loadfdt; then bootz ${loadaddr} - ${fdt_addr}; else bootz; fi; 
mmcdev=0
mmcpart=1
mmcroot=/dev/mmcblk0p2 rootwait rw quiet
netargs=setenv bootargs console=${console},${baudrate} root=/dev/nfs 
netboot=echo Booting from net ...; run netargs; if test ${ip_dyn} = yes; then setenv get_cmd dhcp; else setenv get_cmd tftp; fi; ${get_cmd} ${image}; if test ${boot_fdt} = yes || test ${boot_fdt} = try; then if ${get_cmd} ${fdt_addr} ${fdt_file}; then bootz ${loadaddr} - ${fdt_addr}; else if test ${boot_fdt} = try; then bootz; else echo WARN: Cannot load the DT; fi; fi; else bootz; fi;
ubiroot=ubi.mtd=0 root=ubi0:rootfs0 rootfstype=ubifs rw gpmi
ubiargs=setenv bootargs console=${console},${baudrate} ${ubiroot} consoleblank=0 vt.global_cursor_default=0
loadnorimage=sf read ${loadaddr} 0x000E0000 0x5B8D80; sf read ${fdt_addr} 0x000A0000 0x00040000
norboot=echo Booting from nor ...;run ubiargs; bootz ${loadaddr} - ${fdt_addr}
